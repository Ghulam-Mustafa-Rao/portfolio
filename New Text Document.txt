    <script>
      document.querySelectorAll(".gallery").forEach(gallery => {
        const folder = gallery.dataset.folder;
        if (!folder) return;

        let i = 0;
        function loadNext() {
          const img = new Image();
          img.src = `${folder}/${i}.jpeg`;

          img.onload = () => {
            const div = document.createElement("div");
            div.className = "screenshot";
            div.appendChild(img);

            const videoBlock = gallery.querySelector(".video");
            if (videoBlock) {
              gallery.insertBefore(div, videoBlock);
            } else {
              gallery.appendChild(div);
            }

            i++;
            loadNext();
          };

          img.onerror = () => {
            console.log(`Finished loading images from ${folder}`);
          };
        }

        loadNext();
      });
    </script>
